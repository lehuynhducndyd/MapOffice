package com.example.mapoffice.screens.main

import androidx.compose.animation.AnimatedVisibility
import androidx.compose.foundation.background
import androidx.compose.foundation.layout.Arrangement
import androidx.compose.foundation.layout.Box
import androidx.compose.foundation.layout.Column
import androidx.compose.foundation.layout.Row
import androidx.compose.foundation.layout.fillMaxHeight
import androidx.compose.foundation.layout.fillMaxSize
import androidx.compose.foundation.layout.fillMaxWidth
import androidx.compose.foundation.layout.padding
import androidx.compose.foundation.layout.size
import androidx.compose.foundation.layout.width
import androidx.compose.foundation.lazy.LazyRow
import androidx.compose.material3.Button
import androidx.compose.material3.CircularProgressIndicator
import androidx.compose.material3.FloatingActionButton
import androidx.compose.material3.Scaffold
import androidx.compose.material3.Text
import androidx.compose.material3.pulltorefresh.PullToRefreshBox
import androidx.compose.runtime.Composable
import androidx.compose.runtime.LaunchedEffect
import androidx.compose.runtime.collectAsState
import androidx.compose.runtime.getValue
import androidx.compose.runtime.mutableStateOf

import androidx.compose.ui.Modifier

import androidx.compose.ui.graphics.Color

import androidx.compose.ui.platform.LocalWindowInfo
import androidx.compose.ui.unit.dp
import androidx.compose.ui.zIndex
import androidx.navigation.NavController
import com.example.mapoffice.data.LoadStatus
import com.example.mapoffice.ui_component.CurrentRoomDetail
import com.example.mapoffice.ui_component.RoomList
import org.koin.compose.viewmodel.koinViewModel

@Composable
fun MainScreen(
    navController: NavController,
    viewModel: MainViewModel = koinViewModel()
) {
    val state by viewModel.uiState.collectAsState()
    LaunchedEffect(Unit) {
        viewModel.getRooms()
    }
    val density = LocalDensity.current
    val screenWidth = with(density) {
        LocalWindowInfo.current.containerSize.width.toDp()
    }
    var addRoomPopUpOpen by remember { mutableStateOf(false) }
    var editRoomPopUpOpen by remember { mutableStateOf(false) }
    var deleteRoomPopUpOpen by remember { mutableStateOf(false) }
    Scaffold(
        floatingActionButton = {
            LazyRow(
            ) {
                item {
                    Row(
                        horizontalArrangement = Arrangement.spacedBy(8.dp)
                    ) {

                        Button(
                            onClick = {
                                navController.navigate("group")
                            },
                        ) { Text("Manage Group") }
                        Button(
                            onClick = {
                                if (state.currentRoom == null) return@Button
                                navController.navigate("canvas/${state.currentRoom?.id}")
                            },
                            enabled = state.currentRoom != null
                        ) { Text("Goto Canvas") }
                        Button(
                            onClick = {
                                deleteRoomPopUpOpen = true
                                addRoomPopUpOpen = false
                                editRoomPopUpOpen = false

                            },
                            enabled = state.currentRoom != null
                        )
                        { Text("Delete") }
                        Button(
                            onClick = {
                                editRoomPopUpOpen = true
                                addRoomPopUpOpen = false
                                deleteRoomPopUpOpen = false

                            },
                            enabled = state.currentRoom != null
                        )
                        { Text("Edit") }
                        Button(
                            onClick = {
                                addRoomPopUpOpen = true
                                editRoomPopUpOpen = false
                                deleteRoomPopUpOpen = false
                            },
                        ) { Text("Add") }
                    }
                }

            }
        }
    ) { paddingValue ->
        val isRefreshing = state.loadStatus is LoadStatus.Loading
        PullToRefreshBox(
            isRefreshing = isRefreshing,
            onRefresh = {
                // Gọi hàm tải lại dữ liệu từ API
                viewModel.getRooms()
            },
            modifier = Modifier.fillMaxSize()
        ) {
            Box(
                modifier = Modifier.fillMaxSize().padding(paddingValue)
            ) {
                if (screenWidth < 600.dp) {
                    Column(
                        modifier = Modifier.fillMaxSize(),
                    ) {
                        RoomList(
                            modifier = Modifier.weight(1f).background(Color(0xfff1f1f3))
                                .fillMaxWidth()
                                .padding(vertical = 8.dp),
                            viewModel = viewModel
                        )
                        CurrentRoomDetail(
                            modifier = Modifier.weight(1f).background(Color.White).fillMaxWidth()
                                .padding(vertical = 8.dp),
                            viewModel = viewModel
                        )
                    }
                } else {
                    Row(
                        modifier = Modifier.fillMaxSize(),
                    ) {
                        RoomList(
                            modifier = Modifier.weight(38f).background(Color(0xfff1f1f3))
                                .fillMaxHeight()
                                .padding(horizontal = 8.dp),
                            viewModel = viewModel
                        )
                        CurrentRoomDetail(
                            modifier = Modifier.weight(62f).fillMaxHeight()
                                .padding(horizontal = 8.dp),
                            viewModel = viewModel
                        )
                    }
                }
                AnimatedVisibility(
                    visible = addRoomPopUpOpen,
                    modifier = Modifier.align(Alignment.Center)
                        .fillMaxHeight(0.75f)
                        .width(300.dp)
                ) {
                    AddRoomPopUp(
                        onAddNewRoom = { name, width, height ->
                            viewModel.addRoom(name, width, height)
                            addRoomPopUpOpen = false
                        }, onDimiss = { addRoomPopUpOpen = false }
                    )
                }
                AnimatedVisibility(
                    visible = editRoomPopUpOpen,
                    modifier = Modifier.align(Alignment.Center)
                        .fillMaxHeight(0.75f)
                        .width(300.dp)
                ) {
                    EditRoomPopUp(
                        currentRoom = state.currentRoom ?: return@AnimatedVisibility,
                        onEditRoom = { id, name, width, height ->
                            viewModel.updateRoom(id, name, width, height)
                            editRoomPopUpOpen = false

                        }, onDimiss = { editRoomPopUpOpen = false }
                    )
                }
                AnimatedVisibility(
                    visible = deleteRoomPopUpOpen, modifier = Modifier.align(Alignment.Center)
                        .fillMaxHeight(0.25f)
                        .width(300.dp)
                ) {
                    val roomToDelete = state.currentRoom
                    if (roomToDelete != null) {
                        DeleteRoomPopUp(
                            onConfirmDelete = {
                                viewModel.deleteRoom(roomToDelete.id)
                                deleteRoomPopUpOpen = false
                            },
                            onDimiss = { deleteRoomPopUpOpen = false },
                            name = roomToDelete.name
                        )
                    }
                }

            }
        }
    }

}
